IF COL_LENGTH('dbo.Vouchers', 'name') IS NULL
    EXEC('ALTER TABLE dbo.Vouchers ADD name NVARCHAR(255) NULL');

IF COL_LENGTH('dbo.Vouchers', 'description') IS NULL
    EXEC('ALTER TABLE dbo.Vouchers ADD description NVARCHAR(MAX) NULL');

IF COL_LENGTH('dbo.Vouchers', 'min_booking_amount') IS NULL
    EXEC('ALTER TABLE dbo.Vouchers ADD min_booking_amount DECIMAL(18,2) NOT NULL CONSTRAINT DF_Vouchers_MinBookingAmount_Manual DEFAULT(0)');

IF COL_LENGTH('dbo.Vouchers', 'max_discount_amount') IS NULL
    EXEC('ALTER TABLE dbo.Vouchers ADD max_discount_amount DECIMAL(18,2) NULL');

IF COL_LENGTH('dbo.Vouchers', 'start_date') IS NULL
    EXEC('ALTER TABLE dbo.Vouchers ADD start_date DATETIME2 NULL');

IF COL_LENGTH('dbo.Vouchers', 'end_date') IS NULL
    EXEC('ALTER TABLE dbo.Vouchers ADD end_date DATETIME2 NULL');

IF COL_LENGTH('dbo.Vouchers', 'usage_count') IS NULL
    EXEC('ALTER TABLE dbo.Vouchers ADD usage_count INT NOT NULL CONSTRAINT DF_Vouchers_UsageCount_Manual DEFAULT(0)');

IF COL_LENGTH('dbo.Vouchers', 'is_active') IS NULL
    EXEC('ALTER TABLE dbo.Vouchers ADD is_active BIT NOT NULL CONSTRAINT DF_Vouchers_IsActive_Manual DEFAULT(1)');

IF COL_LENGTH('dbo.Vouchers', 'created_at') IS NULL
    EXEC('ALTER TABLE dbo.Vouchers ADD created_at DATETIME2 NOT NULL CONSTRAINT DF_Vouchers_CreatedAt_Manual DEFAULT(SYSDATETIME())');

IF COL_LENGTH('dbo.Vouchers', 'updated_at') IS NULL
    EXEC('ALTER TABLE dbo.Vouchers ADD updated_at DATETIME2 NOT NULL CONSTRAINT DF_Vouchers_UpdatedAt_Manual DEFAULT(SYSDATETIME())');

IF COL_LENGTH('dbo.Vouchers', 'min_booking_value') IS NOT NULL
AND COL_LENGTH('dbo.Vouchers', 'min_booking_amount') IS NOT NULL
    EXEC('UPDATE dbo.Vouchers SET min_booking_amount = ISNULL(min_booking_value, 0) WHERE min_booking_amount = 0');

IF COL_LENGTH('dbo.Vouchers', 'valid_from') IS NOT NULL
AND COL_LENGTH('dbo.Vouchers', 'start_date') IS NOT NULL
    EXEC('UPDATE dbo.Vouchers SET start_date = valid_from WHERE start_date IS NULL');

IF COL_LENGTH('dbo.Vouchers', 'valid_to') IS NOT NULL
AND COL_LENGTH('dbo.Vouchers', 'end_date') IS NOT NULL
    EXEC('UPDATE dbo.Vouchers SET end_date = valid_to WHERE end_date IS NULL');

IF COL_LENGTH('dbo.Vouchers', 'name') IS NOT NULL
AND COL_LENGTH('dbo.Vouchers', 'start_date') IS NOT NULL
AND COL_LENGTH('dbo.Vouchers', 'end_date') IS NOT NULL
AND COL_LENGTH('dbo.Vouchers', 'is_active') IS NOT NULL
AND COL_LENGTH('dbo.Vouchers', 'usage_count') IS NOT NULL
AND COL_LENGTH('dbo.Vouchers', 'created_at') IS NOT NULL
AND COL_LENGTH('dbo.Vouchers', 'updated_at') IS NOT NULL
    EXEC('
        UPDATE dbo.Vouchers
        SET
            name = ISNULL(NULLIF(LTRIM(RTRIM(name)), ''''), code),
            start_date = ISNULL(start_date, ''2025-01-01''),
            end_date = ISNULL(end_date, ''2026-12-31''),
            is_active = ISNULL(is_active, 1),
            usage_count = ISNULL(usage_count, 0),
            created_at = ISNULL(created_at, SYSDATETIME()),
            updated_at = ISNULL(updated_at, SYSDATETIME())
    ');

IF COL_LENGTH('dbo.Vouchers', 'discount_type') IS NOT NULL
    EXEC('
        UPDATE dbo.Vouchers
        SET discount_type =
            CASE
                WHEN UPPER(discount_type) IN (''PERCENT'', ''PERCENTAGE'') THEN ''Percentage''
                WHEN UPPER(discount_type) IN (''FIXED_AMOUNT'', ''FIXED'') THEN ''Fixed''
                ELSE discount_type
            END
    ');

IF EXISTS (
    SELECT 1
    FROM sys.columns
    WHERE object_id = OBJECT_ID('dbo.Vouchers')
      AND name = 'name'
      AND is_nullable = 1
)
    EXEC('ALTER TABLE dbo.Vouchers ALTER COLUMN name NVARCHAR(255) NOT NULL');

IF EXISTS (
    SELECT 1
    FROM sys.columns
    WHERE object_id = OBJECT_ID('dbo.Vouchers')
      AND name = 'start_date'
      AND is_nullable = 1
)
    EXEC('ALTER TABLE dbo.Vouchers ALTER COLUMN start_date DATETIME2 NOT NULL');

IF EXISTS (
    SELECT 1
    FROM sys.columns
    WHERE object_id = OBJECT_ID('dbo.Vouchers')
      AND name = 'end_date'
      AND is_nullable = 1
)
    EXEC('ALTER TABLE dbo.Vouchers ALTER COLUMN end_date DATETIME2 NOT NULL');
